QUESTION:

Write the program using C

 

ANSWER:

#include <stdio.h>

struct Ingredient
{
    char name[20];
    char quantity[20];
    char measurementType[20];
    
};
struct Recipe
{
    char name[25];
    struct Ingredient ingredients[5]; // array of 5 ingredients 
    
    
};
int main(void) {
    

//initialize struct Recipe
    struct Recipe r = {"Cake",{{"flour","2 cups","pounds"}, {"eggs","4","number"},{"butter","1 cups","pounds"}, {"vanilla","1 spoon","mg"}, {"baking soda","1 spoon","mg"}}};
    
    // display recipe contents 
    printf("Recipe : %s\nIngredients : ",r.name);
    for(int j=0;j<5;j++)
    printf("\n%s  %s in %s ",r.ingredients[j].name,r.ingredients[j].quantity,r.ingredients[j].measurementType);
    
    
    return 0;
}
Output:

Recipe : Cake
Ingredients : 
flour  2 cups in pounds 
eggs  4 in number 
butter  1 cups in pounds 
vanilla  1 spoon in mg 
baking soda  1 spoon in mg
