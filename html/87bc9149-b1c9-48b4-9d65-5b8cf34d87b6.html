public interface Vehicle {
    void Start();
    void Stop();
    void ChangeSpeed();
}

public abstract class Car implements Vehicle{
    int year,make,speed;
    static int count;
    public Car(int year, int make) {
        this.year = year;
        this.make = make;
        this.speed=0;
        count++;
    }
    
    abstract void sound();

    public static int getCount() {
        return count;
    }
    
}

public class SportsCar extends Car{
    String roof;
    int doors;
    
    @Override
    void sound() {
        System.out.println("broooom");
    }

    @Override
    public void Start() {
        System.out.println("Engine Started");
    }

    @Override
    public void Stop() {
        System.out.println("Engine Stopped");
    }

    @Override
    public void ChangeSpeed() {
        speed+=20;
        if(speed>=65){
            try {
                throw new Exception("speed must be under 65");
            } catch (Exception ex) {
                System.out.println(ex.getMessage());
                speed=64;
            }
        }
    }

    public SportsCar(String roof, int doors, int year, int make) {
        super(year, make);
        this.roof = roof;
        this.doors = doors;
    }
    
}
