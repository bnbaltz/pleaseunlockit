#include<iostream>

using namespace std;

 

 

const double KM_TO_MILES=0.6;

const double KG_TO_POUND=2.2;

 

void to_fahrenheit(){

                double celsius;

                cout<<"Enter temperature in celsius: "; cin>>celsius;

                double f_temp= 9*celsius/5+32;

                cout<<"Temperature in Fahrenheit scale is : "<<f_temp<<endl;

}

 

void to_miles(){

                double km;

                cout<<"Enter distance in kilometers: "; cin>>km;

                cout<<"Distance in miles is: "<<km*KM_TO_MILES<<endl;

}

 

void to_pounds(){

 

                double kg;

                cout<<"Enter weight in kilogram: "; cin>>kg;

                cout<<"Weight in pounds is: "<<kg*KG_TO_POUND<<endl;

               

}

 

int main(){

               

                int choice;

                do{

                                cout<<"Converted Toolkit\n"

                                                <<"-------------------------\n"

                                                <<"\t1. Temperature Converter\n"

                                                <<"\t2. Distance Converter\n"

                                                <<"\t3. Weight Converter\n"

                                                <<"\t4. Quit\nEnter Choice:";

                                                cin>>choice;

                                                switch(choice){

                                                                case 1: to_fahrenheit();break;

                                                                case 2: to_miles();break;

                                                                case 3: to_pounds();break;

                                                                case 4: break;

                                                                default: cout<<"Invalid selection\n";

                                                }

                                               

                }while(choice!=4);

               

                cout<<"Bye!\n";

}
