import matplotlib.pyplot as plt #imports matplot
import numpy as np  #imports numpy
from PIL import Image #imports Image from PIL to read and open image
sizee=int(input("Enter Size:"))  #enter the size  of the image
fname=str(input("Enter File Name:")) #enter the name of the image
imgg=np.ones((sizee,sizee,3)) #creates an white pixelled image
for i in range(0,sizee-1,2):  #row loop runs from 0 to size-1 with step size 2
    for j in range(0,sizee): #column loop runs from 0 to size of the image
        imgg[i][j][0]=1 # RGB values of even rows will be set to 101 to create purple
        imgg[i][j][1]=0
        imgg[i][j][2]=1
        imgg[i+1][j][0]=0 #RGB values of odd rows will be set to 011 to create cyone
        imgg[i+1][j][1]=1
        imgg[i+1][j][2]=1
plt.imsave(fname,imgg) #image will be saved as fname
im=Image.open(fname)
im.show() #created image will be opened
