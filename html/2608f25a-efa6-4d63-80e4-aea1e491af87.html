import cv2 #importing cv2 package. cv2 is used for image processing.
import numpy as np #importing numpy package

size = int(input("Enter the image size")) # Taking input from user.
img_pixels_size = size*96 # Converting image size in pixels size
whiteblankimage = 255 * np.ones(shape=[img_pixels_size, img_pixels_size, 3], dtype=np.uint8) # generating blank image with given input size

for i in range(0,img_pixels_size,40):
    cv2.line(whiteblankimage, pt1=(0,i), pt2=(img_pixels_size,i), color=(128,0,128), thickness=20) # Coloring a purple line with thickness of 20 after every loop
    cv2.line(whiteblankimage, pt1=(0,i+20), pt2=(img_pixels_size,i+20), color=(255,255,0), thickness=20, lineType=cv2.LINE_AA) # Coloring a cyan line with thickness of 20 after every loop

cv2.imwrite('stripes.jpg',whiteblankimage) # saving colored striped image in JPG format.
