bool IsDecimalInteger(char str[]) {
    
    // First finding number of characters in string(char array)
    
    int length = strlen(str);
    
    // If length of array is 0, return false
    if(length == 0)
        return false;

    // Then iterating over array and checking if each char is a digit or not
    // Then if any character in array is number or symbol - or +
    // Adding that and if it matches length, return 1
    
    int count = 0;
    int i=0;
    
    for (; i<length; i++) {
        
        // If digit or symbol + or - only at first character(place)
        
        if((str[i] >= '0' && str[i] <= '9') || str[0] == '+' || str[0] == '-')
            count++;
    }
    
    if(count == length)
        return true;
    else
        return false;
}
