I have created the above web page using html for pizza ordering. I have also added the javascript code for validation and calculating total cost of the order and display it to the user. You may add more styling according to your choice.

Note: You may look the screenshot of the code to have better understanding of the indentation.

Html+javascript code:

<!DOCTYPE HTML>

<head>

    <script>
    
    function validate()
    {
        var cost = 0;
        var size = document.forms["order_form"]["size"].value;
        if(size == "small")
            cost = 10;
        else if(size == "medium")
            cost = 12;
        else 
            cost = 14;
        
        //add cost for type
        var type = document.forms["order_form"]["type"].value;
        //if veg lover, do nothing
        //if pepperoni, add 10 to base price
        if(type == "pepperoni")
            cost = cost + 10;
        //if cheese, add 20 to base
        else if(type == "cheese")
            cost = cost + 20;
        
        //add cost for drink
        var drink = document.forms["order_form"]["drink"].value;
        //if any drink is selected, then add $2 to the cost
        if(drink != null)
            cost = cost + 2;
        
        //calculate tax
        var tax = cost*0.13;
        //total amount
        var totalCost = cost+tax;
        
        //display cost to the user
        document.getElementById("invoice").innerHTML = "Order Invoice";
        document.getElementById("cost").innerHTML = "Cost (before tax) = $"+cost;
        document.getElementById("tax").innerHTML = "Tax = $" + tax;
        document.getElementById("totalCost").innerHTML = "Total amount = $" + totalCost;
        document.getElementById("thanks").innerHTML = "Thanks for the order";
    }
    </script>

    }
