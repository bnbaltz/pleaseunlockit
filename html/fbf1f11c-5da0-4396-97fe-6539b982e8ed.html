%problem7_5_4b

%data

gama=1.4

R=0.287

Ta=220

int=1

M=3

Q=45000

cp=1

%determine the range of maximum temperature

T4=2000:int:3000;

%calculate the air velocity

u=M*sqrt(gama*R*Ta);

%calculate the exhaust velocity;

ue=u*sqrt(T4/Ta)*(1+(gama-1)/2*M^2)^(-1/2);

%calculate the fuel consumption ratio

f=(T4-Ta*(1+(gama-1)/2*M^2))./(Q/cp-T4);

%calculate the specific thrust

Thr=u.*((1+f).*sqrt(T4/Ta)*(1+(gama-1)/2*M^2)^(-1/2)-1);

%calculate the specific kinetic energy:

Ke=(1+f).*ue.^2/2-u^2/2;

%calculate the thermal efficiency

thoth=Ke./(f*Q);

%calculate the propeller efficiency

thop=Thr*u./Ke;

%calculate the overall efficiency

thoo=thoth.*thop;

plot(T4,thoth, 'b')

hold on

plot(T4,thop,'k')

hold on

plot(T4,thoo, 'r')
