QUESTION:

JAVA CODING based on Inheritance: -

Create a class Book with the following members: Accession number, Name of the author, Title of the book, Year of Publication, Publisher’s Name, Cost of the book and Availability

Create a class Membership with the following members: Member id, Name, Maxbooks Derive a class Library information_system from Book and Member. Include the following member functions:
1. Issue a book after checking member id, book id, Availability of the book and Maxbooks available for the member.

2. Update availability of book and Maxbooks after each transaction.

3. Display a member’s information given member id

Instructions:-

● Please write approach with 2-3 sentences

● Please add screen shot of output

ANSWER:

Solution : it contains 3 java classes as given below. Create Two text files members.txt and books.txt which contain information about members and books as shown here. 

Book.java 


package library_information_system;

import java.io.BufferedReader;
import java.io.BufferedWriter;
import java.io.File;
import java.io.FileReader;
import java.io.FileWriter;
import java.io.IOException;


public class Book {
    public int accession_number ;
    public String author_name;
    public String title;
    public int year_of_publication;
    public String publisher_name;
    public double cost;
    public boolean availability;
    
    public static Book[] fetchAllBooksFromFile(){
        Book books[] = new Book[10];
        BufferedReader br = null;
      
        try{
           
           br = new BufferedReader(new FileReader("books.txt"));
           for(int i =0; i <10; i++){
                String str= br.readLine();
                String f[]= str.split(" ");   
                books[i] = new Book();
                books[i].accession_number = Integer.parseInt(f[0]);
                books[i].author_name = f[1];
                books[i].title = f[2];
                books[i].year_of_publication = Integer.parseInt(f[3]);
                books[i].publisher_name = f[4];
                books[i].cost = Double.parseDouble(f[5]);
                books[i].availability = Boolean.parseBoolean(f[6]);
           }
           
        } 
        catch (IOException ioe) 
        {
        } 
       finally 
       {
       try {
          if (br != null)
         br.close();
       } 
       catch (IOException ioe) 
           {
        System.out.println("Error in closing the BufferedReader");
       }
    }
 
      return books; 
    }
    
    public static void writeAllBooksInFile(Book[] books){
         try {
            File file = new File("books.txt");
        // if file doesnt exists, then create it
        if (!file.exists()) {
        file.createNewFile();
        }
            FileWriter fw = new FileWriter(file.getAbsoluteFile());
            try (BufferedWriter bw = new BufferedWriter(fw)) {
                String str;
                for(int i=0; i<books.length; i++){
                    str =  books[i].accession_number + " "+books[i].author_name +" "+ books[i].title+" "+books[i].year_of_publication+" "+books[i].publisher_name +" "+ books[i].cost+" "+books[i].availability;
                    bw.append(str);
                    bw.newLine();
                }
            }
        } catch (IOException e) {
            e.printStackTrace();
        }
    }
}

package library_information_system;

import java.util.Scanner;

public class Library_Information_System {
    
    public static boolean issueBook(int bookId, int memberId){
        //get all information of books and members to know if given ids are valid or not 
        Book books[] = Book.fetchAllBooksFromFile();
        Membership membership[] = Membership.fetchAllMembersFromFile();
        // find book with given book id 
        Book b = new Book();
        Membership m = new Membership();
        boolean flag= false;
        for(int i=0 ; i<books.length; i++){
            if(books[i].accession_number == bookId){
                b = books[i];
                flag = true;
                break;
            }
        }
        if(flag == false){
            System.out.println("No book found with given Id");
        }
        if(b.availability == false){
            System.out.println("Book is unavilable");
        }
        // find member with given member id. 
        flag = false;
        for(int i=0; i<membership.length; i++){
            if(membership[i].memberId == memberId){
                m = membership[i];
                flag = true;
                break;
            }
        }
        if(flag == false){
            System.out.println("No Member found with given Id");
        }
        if(m.maxbooks <=0){
            System.out.println("Member Can not issue more books");
        }
        if(b.availability && m.maxbooks>0){
            return true;
        }
        else {
            return false;
        }
    }
 
    public static void updateAvailability(int bookId, int membershipId){
        //get all information of books and members to know if given ids are valid or not 
        Book books[] = Book.fetchAllBooksFromFile();
        Membership membership[] = Membership.fetchAllMembersFromFile();
        // find book with given book id 
        Book b = new Book();
        Membership m = new Membership();
        for(int i=0 ; i<books.length; i++){
            if(books[i].accession_number == bookId){
                b = books[i];
                b.availability = false;
                break;
            }
        }
        //update book list in file 
        Book.writeAllBooksInFile(books);
        // find member with given member id. and update it 
        for(int i=0; i<membership.length; i++){
            if(membership[i].memberId == membershipId){
                m = membership[i];
                m.maxbooks = m.maxbooks -1;
                break;
            }
        }
        //update member list in file 
        Membership.writeAllMembershipsInFile(membership);
        
        System.out.println("Book has been issued ");
    }
    
    public static void getMemberInformation(int membershipid){
        //get all members 
        Membership membership[] = Membership.fetchAllMembersFromFile();
        Membership m = new Membership();
        boolean flag = false;
        for(int i=0; i<membership.length; i++){
            if(membership[i].memberId == membershipid){
                m = membership[i];
                flag = true;
                break;
            }
        }
        if(flag == false){
            System.out.println("No Member found with given Id");
            return;
        }
        else {
            System.out.println("\n--------------------------------\nMember information with id "+ membershipid);

            System.out.println("Max Books can be issued by member :"+m.maxbooks);

    } 
    
    public static void main(String[] args) {
        boolean flag = true, available ;

                         break;
                case 3 : flag =false ; break;
                default : System.out.println("Choose correct option "); break;

        }
        
    }
    
}
