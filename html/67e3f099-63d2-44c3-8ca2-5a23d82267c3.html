#include<iostream>
#include<fstream>

using namespace std;

int main(){
    const int SIZE = 10;
    int arr[SIZE],swaps = 0;
    for(int i=0;i<SIZE;i++){
        cout<<"Enter slot "<<i<<": ";
        cin >> arr[i];
    }
    
    for(int i=0; i<SIZE;i++){
        for(int j=0; j<SIZE-i-1;j++){
            if(arr[j]>arr[j+1]){
                int t = arr[j];
                arr[j] = arr[j+1];
                arr[j+1] = t;
                swaps++;
            }
        }
        //display the array
        for(int i=0;i<SIZE;i++)cout<<arr[i]<<"|";
        cout<<"Number of swaps: "<<swaps<<endl;
        
    }
    
    ofstream out("output.csv");
    if(!out){
        cout<<"Error: could not write data to file.";
        return 1;
    }
    
    for(int i=0;i<SIZE;i++)out<<arr[i]<<",";
    out.close();
    cout<<"\n\ndata successfully written to file ...";

    return 0;
}
