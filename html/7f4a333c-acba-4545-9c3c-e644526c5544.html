Solution:

file=input()  #read input file name from user
f=open(file,"r")  #open in read mode
data=list(f.readlines())  #store data in file in list
count=0
total=0

for i in range(len(data)):   #For every line of data in list
    data[i]=data[i][:-1]  #remove new line character


for i in data:
    temp=i.split(" ")
    
    if int(temp[-1])>5000:   #In each record if value is greater than 5000, then increment count
        count+=1
        total+=int(temp[-1])   #calculate total each time
new_file="bytes_"+file    #create new file for storing output data
f=open(new_file,"w")    #open the file in write mode
f.write(str(count))    #store the output result in file
f.write("\n")
f.write(str(total))
f.close()    #close the file
