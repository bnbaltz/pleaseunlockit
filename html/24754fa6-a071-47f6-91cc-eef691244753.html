#include <iostream>
#include <cmath>//if cmath does not work use math.h

using namespace std;

int solution(int M, int N){
    int areaOfM=1*1; //calculating area of M tile
    int areaOfN=2*2; //calculating area of N tile
    int totalArea=M*areaOfM+N*areaOfN; //calculating total area by combining total number of both tiles
    totalArea=floor(sqrt(totalArea));//Here we are taking sqaure root and then flooring it...this will give side of maximum sqaure
    return totalArea;//returning area
}
int main()
{
    //Testing it with different values
    cout<<"M=8 N=0-->"<<solution(8,0)<<endl;
    cout<<"M=4 N=3-->"<<solution(4,3)<<endl;
    cout<<"M=0 N=18-->"<<solution(0,18)<<endl;
    cout<<"M=13 N=3-->"<<solution(13,3)<<endl;
    return 0;
}
