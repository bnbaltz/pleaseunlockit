public class ElectionTesterV1{

    // method to print candidates
    public static void printCandidates(Candidate[] arr){
        // for each candidate print candidate
        for(Candidate c: arr){
            System.out.println(c);
        }
    }

    // method to count votes
    public static int countTotalVotes(Candidate[] arr){
        int totalVotes = 0;
        // for each candidates add its vote in total votes and return total votes
        for(Candidate c:arr){
            totalVotes += c.getVotes();
        }
        return totalVotes;
    }

    // method to print the table 
    public static void printTable(Candidate[] arr){
        // print the header
        System.out.format("%-20s%-20s%-20s","candidate","Votes Recieved","% of total votes");
        System.out.println("\n===========================================================");

        // store total votes
        int totalVotes = countTotalVotes(arr);
        // for each candidate print required info using getters of candidate
        for(Candidate c: arr){
            System.out.format("%-25s%-20s%-20.2f",c.getName(),c.getVotes(),(double)c.getVotes()*100/totalVotes);
            System.out.println();

        }

        // print total votes
        System.out.println("\nThe total number of votes is: "+totalVotes);
    }


    // main method which test this
    public static void main(String [] args){

        // make arr of 5 candidate
        Candidate arr[] = new Candidate[5];

        // assign values 
        arr[0] = new Candidate("Tony Stark",3691);
        arr[1] = new Candidate("Henry Pym",2691);
        arr[2] = new Candidate("B____ B____",1962);
        arr[3] = new Candidate("S____ R____",1491);
        arr[4] = new Candidate("C____ D_____",1968);

        // print data
        System.out.println("Raw Election Data:\n");
        printCandidates(arr);
        System.out.println("\nElection results: \n");
        printTable(arr);
        

    }
}

            System.out.println();

        }

        System.out.println("\nThe total number of votes is: "+totalVotes);
    }

    // method to change name
    public static void changeName(Candidate[] arr, String name, String newName){
        // traverse through index and if the name found set its name to new name by setter of candidate
        for(int i=0;i<arr.length;i++){
            if (arr[i].getName() == name){
                arr[i].setName(newName);
            }
        }
    }

    // method to change votes, same as above
    public static void changeVotes(Candidate[] arr, String name, int newVote){
        for(int i=0;i<arr.length;i++){
            if (arr[i].getName() == name){
                arr[i].setVotes(newVote);
            }
        }
    }
    // method to change both name and votes, combine both
    public static void changeNameAndVote(Candidate[] arr, String name, String newName, int newVote){
        for(int i=0;i<arr.length;i++){
            if (arr[i].getName() == name){
                arr[i].setName(newName);
                arr[i].setVotes(newVote);
            }
        }
    }
