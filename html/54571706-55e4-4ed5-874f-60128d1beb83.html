//import statements

import java.util.Scanner;

// Searching is a very basic problem. given a list of obejct we are required to find the position od the object in that list
// Eg:- [1,3,4,2,5,6] is a list and we need to find the position of 4. The searching algorithm returns 2 if the list is 0-indexed.


//class defination
public class Searching {
    // function to perform linear search and print the search indices
    public void linear_search(int A[], int target) {
        // in linear search we explore the array elements sequentially and match it with the target element
        // we do not need the array to be sorted for this search technique.
        // if a match is found we stop there and return the index where the element was found
        for (int i = 0; i < A.length; i++) {
            System.out.print(i + " ");
            if (target == A[i]) { // checking if the target is equal to the ith element in A
                break; // since the program does not requre us to return we simppy break out of the loop
            }
        }
        System.out.println();
    }

    // function to perform binary search and print the search indices
    public void binary_search(int A[], int target) {
        // binary search is a divide-and-conquer approach which requires the array to be sorted
        // we first search the middle of the array.
        // If the middle element is smaller we are sure that the required element is between middle+1th and last element.
        // If the middle element is larger we are sure that the required element is between middle-1th and first element.
        // after every step we discard half of the remaining array elements
        int lo = 0;
        int hi = A.length - 1;
        while (lo <= hi) {
            int mid = (lo + hi) / 2; // calculate the middle element
            System.out.print(mid + " ");
            if (A[mid] == target) {
                break; // if the middle element is our target then break
            } else if (A[mid] < target) {
                lo = mid + 1; // if middle element < target, search only middle+1 to last element
            } else {
                hi = mid - 1; // if middle element > target, search only middle-1 to first element
            }
        }
        System.out.println();
    }

    // driver function to input the values and perform the search operation
    public static void main(String[] args) {

        System.out.println("SEARCHING : Enter 15 numbers in ascending order");
        int[] A = new int[15]; // array of size 15
        Scanner sc = new Scanner(System.in); // scanner class used to take input
        for (int i = 0; i < 15; i++) { // input 15 integers
            System.out.print("Enter slot " + i + ": ");
            A[i] = sc.nextInt();
        }
        System.out.print("Enter the target: ");
        int target = sc.nextInt(); // input the target to be searched
        for (int i = 0; i < A.length; i++) {
            System.out.print(A[i] + "|"); // print the sorted array entered
        }
        System.out.println();
        Searching S = new Searching(); // create object of searching class
        S.linear_search(A, target); // call linear search using the object
        S.binary_search(A, target); // call binary search using the object
    }
}
