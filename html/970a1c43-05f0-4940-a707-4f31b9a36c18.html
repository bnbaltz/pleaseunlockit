Here is the code for the printing the david star in java using loops:

You can use the same logic for python changing the syntax

static void Triangle(int nBase)
{
    int i, j, move;
    for (i = 1; i <= nBase; i++)
    {
        move = 0;
        do
        {
            Console.Write(" ");
            move++;
        } while (move <= nBase * 3 - i + 1);
        for (j = 1; j <= i; j++)
        {
            Console.Write("* ");
        }
        Console.WriteLine();
    }
}
static void TriangleInverted(double nBase)
{
    int i, j, move;
    for (i = (int)nBase; i >= 0; i--)
    {
        move = 0;
        do
        {
            Console.Write(" ");
            move++;
        } while (move <= nBase * 3 - i + 1);
        for (j = 1; j <= i; j++)
        {
            Console.Write("* ");
        }
        Console.WriteLine();
    }
}
