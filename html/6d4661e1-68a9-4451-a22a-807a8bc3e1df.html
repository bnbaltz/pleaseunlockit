PYTHON CODE :

lis = list(map(int,input().split())) #input elements
lst1 = []
count = 1
for i in range(0, len(lis) - 1):
    if lis[i] == lis[i + 1]: # checking the adjacent elements
        count = count + 1
    else:
        lst1.append(count) # append the count to the list
        count = 1
lst1.append(count)
print(max(lst1)) # print the max count of adjacent elements
