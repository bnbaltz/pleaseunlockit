import random

#function to determine current position of hare
def hare_position(pos):
    next_pos = random.choices(['s','bh','bs','sh','ss'],weights = (0.2,0.2,0.1,0.3,0.2))

    if next_pos == ["s"]:
        return pos
    elif next_pos == ["bh"]:
        return min(pos+7,50)
    elif next_pos == ["bs"]:
        return max(1,pos-10)
    elif next_pos == ["sh"]:
        return min(pos+1,50)
    elif next_pos == ["ss"]:
        return max(pos-2,1)

#function to determine current position of tortoise

    next_pos = random.choices(["fp","s","sp"],weights = (0.5,0.2,0.3))

    elif next_pos == ["fp"]:
        return min(pos+3,50)
    elif next_pos == ["sp"]:
        return min(pos+1,50)
