import java.util.Scanner;
import java.io.File;
import java.io.FileNotFoundException;
public class FileRead{
    public static void main(String args[])throws FileNotFoundException {
        Scanner scanner = new Scanner(new File("input.txt"));
        int id,won,lost,count=0,best_team=0;
        double average,best_average=0;
        
        while(scanner.hasNextInt())
        {
            id = scanner.nextInt();
            won = scanner.nextInt();
            lost = scanner.nextInt();
            count++;
            
            if(won+lost>25) {
                System.out.println("Invalid input (win + lost): "+(won+lost));
            }
            else {
                System.out.println("Team : "+id);
                System.out.printf("%d wins\t%d Losses\n",won,lost);
                if(won+lost==25) {
                    System.out.println("Total game Played : "+(won+lost)+"\tSeason ended");
                }
                else {
                    System.out.println("Total game Played : "+(won+lost)+(25 - (won+lost))+"\t games remaining");
                }
                average = won/(won+lost);
                if(average>best_average) {
                    best_average = average;
                    best_team = id;
                }
                System.out.println("The Winning Average : "+ average);
                
                if(25 - (won+lost)>=won) {
                    System.out.println("The number of game remaining is greater than or equal to number of game won");
                }
                else {
                    System.out.println("The number of game remaining is lesser than number of game won");
                }
                if(25 - (won+lost)>=lost) {
                    System.out.println("The number of game remaining is greater than or equal to number of game lost");
                }
                else {
                    System.out.println("The number of game remaining is lesser than number of game lost");
                }
            }
            
        }
        System.out.println("Total game played : "+count);
        System.out.println("Best team : "+best_team +" with Average of "+ best_average);
                
        
        scanner.close();
    }
}
