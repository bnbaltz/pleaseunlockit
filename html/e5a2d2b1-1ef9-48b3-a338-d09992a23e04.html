import java.util.*;
import java.lang.*;

class Main
{ 
public static void main(String args[])
 {
int[] arr;
 Scanner sc=new Scanner(System.in);
 int T=sc.nextInt();
 arr=new int[T];
 for(int i=0;i<T;i++)
 {
    int N=sc.nextInt(); //Number of students
    int M=sc.nextInt(); //Number of chairs
    if(M>=N)            //If chairs>=students
        arr[i]=0;       //no chairs required
    else                //if chairs<students
        arr[i]=N-M;     //additional chairs is (students-chairs)
 }
 for(int i=0;i<T;i++)
    System.out.println(""+arr[i]);
}
}
