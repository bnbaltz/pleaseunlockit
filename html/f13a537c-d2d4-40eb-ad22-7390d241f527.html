# reading input
N = int(input())
A = list(map(int,input().split()))
k = int(input())
# declaring variable to store minimum contigous usm
min_sum = float('inf')
# finding sum of contigous array
for i in range(N):
    for j in range(N-1,i,-1):
        s = sum(A[i:j+1])
        # if sum less than k ignore
        if s <= k:
            continue
        # if sum is greater than k
        else:
            # checking that the sum greater than k is less than previous sum
            if min_sum > s:
                min_sum = s
# if minimum sum equals infinite than no solution
if min_sum == float('inf'):
    print(-1)
# otherwise printing the result
else:
    print(min_sum)
