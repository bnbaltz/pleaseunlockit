package com.chegg.biotech;

import java.util.Random;

/**
 * Class creates the unique brand name
 * 
 * @author BB
 */
public class BrandName {
	// array containing consonants
	static String[] consonants = { "b", "c", "d", "f", "g", "h", "j", "k", "l",
			"m", "n", "p", "r", "s", "t", "v", "w", "x", "z" };
	
	// array containing vowels
	static String[] vowels = { "a", "e", "i", "o", "u" };
	
	//min and max length for the brandName
	final static int minLength = 6;
	final static int maxLength = 12;

	/**
	 * main method to run the application
	 * 
	 * @param args
	 */
	public static void main(String[] args) {
		
		System.out.println("Brand Name Generator - E. Eckert\n");

		/**
		 * for loop to generate the 10 unique brandName where the length for
		 * name is also generated randomly within the given range of 6 to 12
		 */
		for (int i = 0; i < 10; i++) {
			System.out.println( i+1 +") "+ generateName((new Random()
					.nextInt(maxLength - minLength + 1)) + minLength));
		}
	}

	/**
	 * This method will generate the brand name with below criteria
	 * 1. Brand name string must have length from 6 to 12
	 * 2. Odd Position letter (1,3,5...) must be consonants 'bcdfghjklmnprstvwxz'
	 * 3. Even Position letter (2,4,6...) must be vowels 'aeiou'
	 * 4. Letter Q and Y are avoided as they do not resonate positivety
	 * 5. First letter of Brand Name must be Capital
	 * 
	 * @param nameLength
	 *            is the length of string for brand name
	 * @return String returns the brand name
	 */
	private static String generateName(int nameLength) {
		String brandName = new String();
		for (int j = 1; j <= nameLength; j++) {
			if (j % 2 == 0) {
				brandName += vowels[new Random().nextInt(vowels.length)];
			} else {
				brandName += consonants[new Random().nextInt(consonants.length)];
			}
		}
		return brandName.substring(0, 1).toUpperCase() + brandName.substring(1);
	}
}
