import java.util.Scanner; 

public class Main
{
	public static void main(String[] args) 
	{
	    // Variable declarations.
	    String UserName = "";
        String HighestScoreUserName = "";
        int Score = 0;
        int HighestScore = 0;
        
	    // This class is used to take user input.
	    Scanner s1 = new Scanner(System.in);
	    
	    // Asking the user to provide number of students.
        System.out.print("Enter the number of students: ");
        int numOfStudents = Integer.parseInt(s1.next());
        
        // If number of students provided is greater than 0.
        if(numOfStudents > 0)
        {
            // looping to get the user input for the Student name and marks. 
            for(int i = 1; i <= numOfStudents;i++)
            {
                System.out.println("Student :"+ i);
                System.out.print("Enter Name :");
                UserName = s1.next();
                System.out.print("Enter Score :");
                Score = Integer.parseInt(s1.next());
                
                // Checking if the score is greater than highest score than assigning that student as the student with highest score.
                if(Score > HighestScore)
                {
                    HighestScore = Score;
                    HighestScoreUserName = UserName;
                }
            
                System.out.println("Student with the highest score: " + HighestScoreUserName);
        }
	}
}
}
