QUESTION:

Write a program in C that does the following,...

 â€¢Create 2 nested structures  (structs) to store ingredients for a recipe.

1.Store each ingredient's name, needed quantity, and measurement type (i.e., "cup", "pounds")

2.Store the name of the recipe and an array of ingredient struct instances

Print out the recipe's name and list each of its ingredients including the quantity needed

ANSWER:

#include<stdio.h>
struct Ingredient{
    char* name;
    double quantity;
    char* measurement_type;
};
struct Recipe{
    char* name;
    struct Ingredient ingredients[100]; // max 100 ingredients for a recipe // You can change it to number you want
};

int main(){
    struct Ingredient i1;
    i1.name = "water";
    i1.quantity = 1;
    i1.measurement_type = "cup";
    struct Ingredient i2;
    i2.name = "milk";
    i2.quantity =  1;
    i2.measurement_type = "cup";
    struct Ingredient i3;
    i3.name = "tea powder";
    i3.quantity =  5;
    i3.measurement_type = "spoons";
    struct Recipe r1;
    r1.name = "tea";
    r1.ingredients[0]=i1;
    r1.ingredients[1]=i2;
    r1.ingredients[2]=i3;
    printf("Recipe name : %s\n",r1.name);
    printf("Ingredients : \n");
    printf("\t %.2lf %s of %s \n",i1.quantity,i1.measurement_type,i1.name);
    printf("\t %.2lf %s of %s\n",i2.quantity,i2.measurement_type,i2.name);
    printf("\t %.2lf %s of %s\n",i3.quantity,i3.measurement_type,i3.name);
    
}
