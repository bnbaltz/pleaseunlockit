//Function that extracts domain name
int find_domain (char *s1, char *s2)
{
    int i, lastDotPos, aPos, j=0;
    int len = strlen(s1);

    //Checking length
    if(len < 4)
    {
        s2 = "\0";
        return 0;
    }

    //Position of last dot
    lastDotPos = len-4;

    //Comparing characters
    if( (*(s1+lastDotPos)=='.') && (*(s1+lastDotPos+1)=='e') && (*(s1+lastDotPos+2)=='d') && (*(s1+lastDotPos+3)=='u'))
    {
        i=0;
        aPos=-1;

        //Searching for position of @
        for(i=0; *(s1+i)!='\0'; i++)
        {
            //Checking for @
            if(*(s1+i) == '@')
            {
                aPos = i;
                break;
            }
        }

        //Checking aPos
        if(aPos == -1)
        {
            s2 = "\0";
            return 0;
        }
        else
        {
            //Fetching domain
            for(i=aPos+1; *(s1+i)!='\0'; i++, j++)
            {
                //Copying values
                *(s2+j) = *(s1+i);
            }

            //Adding Null
            *(s2+j) = '\0';
            return 1;
        }
    }
    else
    {
        s2 = "\0";
        return 0;
    }
}


//read_line function
int read_line(char *str, int n)
{
    int ch;
    int i=0;

    while((ch=getchar()) != '\n')
    {
        if(i < n)
        {
            *str++ = ch;
            i++;
        }
    }
    *str = '\0';
    return i;
}
