Create an object called game in the script.js file.

  • Add 2 properties: lives - initially 3, and coins - initially 0.
  • Add a getter called points that returns coins * 10.
  • Add a playerDies() method that subtracts 1 from lives if lives is greater than 0.
  • Add a newGame() method that sets lives to 3 and coins to 0.

The script.js file includes several console.log() statements, which should match the output below if the game object works correctly.

lives = 3
coins = 0
points = 0
points = 20
lives = 2
lives = 0
lives = 3
coins = 0

View the program's output in the browser's JavaScript console.

 

ANSWER:

Code - 

//an object called game 
var game = {};
//Add 2 properties: lives - initially 3, 
game.lives = 3;
//coins - initially 0.
game.coins = 0;
//add a getter called points that returns coins * 10.
Object.defineProperty(game, 'points', {
//getter added name point
  get() {
  //return coins*10
    return this.coins*10;
  }
});
//|Add a playerDies() method that subtracts 1 from lives if lives is greater than 0.
game.playerDies = function(){
//method that subtracts 1 from lives if lives is greater than 0.
    if(this.lives>0){
      this.lives--;
  }
}
//Add a newGame() method that sets lives to 3 and coins to 0.

game.newGame = function(){
//that sets lives to 3 and coins to 0.
    this.lives = 3;
  this.coins = 0;
}
