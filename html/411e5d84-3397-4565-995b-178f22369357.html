import java.util.Random;
import java.util.Scanner;
public class GuessGame
{
   public static boolean getAndCheckGuess(int numberToGuess)
   {
       Scanner keyboard=new Scanner(System.in);
       System.out.println("Enter your guess: ");
       int userGuess;
       userGuess=keyboard.nextInt();
       if(userGuess>numberToGuess)
       {
           System.out.println("Too High");
       }
       else if(userGuess<numberToGuess)
       {
           System.out.println("Too Low");
       }
       else if(userGuess==numberToGuess)
       {
           System.out.println("You Guessed It");
           return(true);
       }
       else if(userGuess<1||userGuess>20)
       {
           System.out.println("Guess is not Valid");
       }
       return(false);
   }  


public static void main(String args[])
{
   int numToGuess=0;
   boolean correctGuess=false;
   Random rand=new Random();
  
   int flag=1,flag1=1;
   Scanner sc=new Scanner(System.in);
   while(flag!=0)
   {
       numToGuess=rand.nextInt(100)+1;
       correctGuess=false;
       flag1=1;
       while(correctGuess==false && flag1!=0)
       {      
           correctGuess=getAndCheckGuess(numToGuess);
           if(correctGuess==false)
           {
               System.out.println("Do you need more guess enter 1/0:");
               flag1=sc.nextInt();
           }
      
       }
   System.out.println("Do you want to generate a new number enter 1/0:");
   flag=sc.nextInt();
   }
}
}
