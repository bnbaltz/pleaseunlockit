/*The following CircularSinglyLinkedList class

* represents a circular singly linked list

* with only a header node and describes the Josephus

problem. */

//CircularSinglyLinkedList class implementation

//Create the class

public class CircularSinglyLinkedList

{

//private Node class implementation

private static class Node

{

//variables for Node class

//declare the necessary variables

public AnyType data;

public Node next;

//parameterized constructor for Node class

public Node(AnyType x)

{

data = x;

next = null;

} //end of the constructor

} //end of Node class

//reference variables for the class

private Node head;

private int theSize;

//default constructor implementation

public CircularSinglyLinkedList()

{

head = null;

theSize = 0;

} //end of default constructor

/* The following playJosephusGame method

* solves the Josephus problem. */

//playJosephusGame method implementation

public void playJosephusGame(int N, int M)

{

//start the game from the first player

Node current = head;

/* verify whether the number of

* players in the circle is 1 */

while(size() > 1)

{

// display the alive players

System.out.println("\nAlive: "

+ toString());

//make M passes in the circle

for(int i = 1; i <= M; i++)

current = current.next;

// display the eliminated player

System.out.println("Killed: "

+ remove(current.data));

// move to the next player

current = current.next;

} //end while

//display the winner of the game

System.out.println("\nWinner of the "

+ "game: " + current.data);

} //end of playJosephusGame method

/* The following add method accepts an element as a

* parameter, and then adds that element

* to the end of the list. */

// add method implementation

public void add(AnyType x)

{

// create a new node with the element

Node newNode = new Node(x);

// verify whether list is empty

if(isEmpty())

// make the new node as the head node

head = newNode;

else

{

// go to the end of the list

Node current = head;

while(current.next != head)

current = current.next;

// add the new node at the end of list

current.next = newNode;

} // end if/else

newNode.next = head;

// increase the size of the list

theSize++;

} //end of add method

/* The following remove method accepts an

* element as a parameter. It removes and returns the

* reference of that element if that element is

* contained in the list, and returns

null otherwise. */

// remove method implementation

public AnyType remove(AnyType x)

{

//return null if the list is empty

if(isEmpty())

return null;

// local variables

Node current = head;

Node prev = head;

while(prev.next != head)

prev = prev.next;

/* repeat the loop for all elements in the list

* or until the element is found in the list */

while(((Comparable)current.data).compareTo(x) != 0)

{

/* return from null if the element

* is not contained in the list */

if(current.next == head)

return null;

prev = current;

current = current.next;

} //end while

/* remove the element from the

* list and update the list */

if(prev.next == head)

{

head = head.next;

prev.next = head;

}

else

{

prev.next = current.next;

} // end if/else

// decrease the size of the list

theSize--;

// return removed element

return current.data;

} // end of remove method

/* The following get method accepts an

* index value as a parameter. It returns null

* if the index is invalid, and returns the

* element located at that index. */

//get method implementation

public AnyType get(int idx)

{

/* return null if the index is invalid */

if(idx < 0 || idx >= size())

return null;

// local variable to refer the current node

Node current = head;

int i = 0;

// repeat the loop to get the node at idx

while(i < idx)

{

// go to the next node

current = current.next;

i++;

} // end loop

// return element located at the index

return current.data;

} // get method implementation

/* The following size method returns the

* number of elements in the list. */

// size method implementation

public int size()

{

return theSize;

} // end of size method

/* The following isEmpty method returns true

* if the list is empty and

* returns false otherwise. */

// isEmpty method implementation

public boolean isEmpty()

{

return size() == 0;

} // end of isEmpty method

/* The following printList method

* displays the elements in the list. */

// printList method implementation

public void printList()

{

//return from the method

//if the list is empty

if(isEmpty())

return;

//repeat the loop for all elements in the list

Node current = head;

while(current.next != head)

{

System.out.print(current.data + " ");

current = current.next;

} // end while

System.out.println(current.data);

} // end of printList method

/* The following toString method returns

* a string representation of the list. */

// toString method implementation

public String toString()

{

// return the empty string

//if the list is empty

if(isEmpty())

return "";

// local variables

String elements = "";

Node current = head;

//repeat the loop for

//all elements in the list

while(current.next != head)

{

elements += current.data + " ";

current = current.next;

} //end while

elements += current.data;

//return a string representation

return elements;

} //end of toString method

} //end of CircularSinglyLinkedList class
