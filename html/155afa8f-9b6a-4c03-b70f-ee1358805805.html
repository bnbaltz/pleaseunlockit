using System; 

class Bubble 
{ 
	static void bubbleSort(int []arr) 
	{ 
		int n = arr.Length; 
		int c=0;
		for (int i = 0; i < n - 1; i++) 
		{	for (int j = 0; j < n - i - 1; j++) 
			   {	if (arr[j] > arr[j + 1]) 
				{ 
					// swap temp and arr[i] 
					int temp = arr[j]; 
					arr[j] = arr[j + 1]; 
					arr[j + 1] = temp; 
					c++;
				
				} 
				
			   }
			    
			    printArray(arr); 
				Console.Write("  Num swaps: " + c);
				Console.WriteLine(); 
			   
		}
	} 

	/* Prints the array */
	static void printArray(int []arr) 
	{ 
		int n = arr.Length; 
		for (int i = 0; i < n; ++i) 
			Console.Write(arr[i] + "|"); 
		//Console.WriteLine(); 
	} 

	// Main ()
	public static void Main() 
	{ 
		
		int []num = new int[10];
		// input from user
		for(int i = 0;i<10;i++)
        {   Console.Write("Enter slot "+ i + " : ");
            num[i]= Convert.ToInt32(Console.ReadLine());
            //Console.WriteLine();
        }
		bubbleSort(num);
		
		 
	} 

}
