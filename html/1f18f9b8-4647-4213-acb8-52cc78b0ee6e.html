import java.util.Scanner;

 

public class DNAPairing {

public static void main(String[] args) {

Scanner input = new Scanner(System.in);

int n = input.nextInt();

String strand, comp_strand;

for (int i = 0; i < n; i++) {

strand = input.next();

comp_strand = "";

for (int j = 0; j < strand.length(); j++) {

if (strand.charAt(j) == 'A')

comp_strand += 'T';

else if (strand.charAt(j) == 'T')

comp_strand += 'A';

else if (strand.charAt(j) == 'G')

comp_strand += 'C';

else if (strand.charAt(j) == 'C')

comp_strand += 'G';

}

System.out.println(strand+"\n"+comp_strand+"\n");

}

}

}
