//KittyMap.java

public class KittyMap {

 

       private boolean[][] kittyGrid;

      

       public KittyMap(int rows, int cols)

       {

             kittyGrid = new boolean[rows][cols];

             for(int i=0;i<rows;i++)

             {

                    for(int j=0;j<cols;j++)

                    {

                           double n = Math.random();

                           if(n < 0.5)

                                 kittyGrid[i][j] = true;

                           else

                                 kittyGrid[i][j] = false;

                    }

             }

       }

      

       public void printKittyCount()

       {

             int kittyCounts[][] = getKittyCountsGrid();

             for(int i=0;i<kittyCounts.length;i++)

             {

                    for(int j=0;j<kittyCounts[i].length;j++)

                           System.out.print(kittyCounts[i][j]+" ");

                    System.out.println();

             }

       }

      

       private int[][] getKittyCountsGrid()

       {

             int [][] kittyCounts = new int[kittyGrid.length][kittyGrid[0].length];

            

             for(int i=0;i<kittyCounts.length;i++)

                    for(int j=0;j<kittyCounts[i].length;j++)

                           kittyCounts[i][j] = getKittyCount(i,j);

             return kittyCounts;

       }

      

       public int getKittyCount(int r, int c)

       {

             if(kittyGrid[r][c])

                    return 9;

             int count = 0;

             if(inBounds(r-1,c-1) && kittyGrid[r-1][c-1])

                    count++;

             if(inBounds(r-1,c) && kittyGrid[r-1][c])

                    count++;

             if(inBounds(r-1,c+1) && kittyGrid[r-1][c+1])

                    count++;

             if(inBounds(r,c-1) && kittyGrid[r][c-1])

                    count++;

             if(inBounds(r,c+1) && kittyGrid[r][c+1])

                    count++;

             if(inBounds(r+1,c-1) && kittyGrid[r+1][c-1])

                    count++;

             if(inBounds(r+1,c) && kittyGrid[r+1][c])

                    count++;

             if(inBounds(r+1,c+1) && kittyGrid[r+1][c+1])

                    count++;

            

             return count;

       }

      

       private boolean inBounds(int r, int c)

       {

             if((r >=0 && r < kittyGrid.length) && (c >=0 && c < kittyGrid[r].length))

                    return true;

             return false;

       }

      

       public String toString()

       {

             String output= "";

             for(int i=0;i<kittyGrid.length;i++)

             {

                    for(int j=0;j<kittyGrid[i].length;j++)

                           if(kittyGrid[i][j])

                                 output += "1 ";

                           else

                                 output += "0 ";

                    output += "\n";

             }

            

             return output.trim();

       }

      

      

}

//end of KittyMap.java
