function setup() {
    createCanvas( 640, 480 );

    // set the frame rate for the sketch
    frameRate( 4 );
}
var canvas = new fabric.Canvas('c', { selection: false });

var line, isDown;

canvas.on('mouse:down', fun(o){
  isDown = true;
  var pointer = canvas.getPointer(o.e);
  var points = [ pointer.x, pointer.y, pointer.x, pointer.y ];
  line = new fabric.Line(points, {
    strokeWidth: 5,
    fill: 'red',
    stroke: 'red',
    originX: 'center',
    originY: 'center'
  });
  canvas.add(line);
});

canvas.on('mouse:move', fun(o){
  if (!isDown) return;
  var pointer = canvas.getPointer(o.e);
  line.set({ x2: pointer.x, y2: pointer.y });
  canvas.renderAll();
});

canvas.on('mouse:up', fun(o){
  isDown = false;
});
