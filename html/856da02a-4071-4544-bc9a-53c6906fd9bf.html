Some of the persons do not want to be understood by others who are around them,

so they start to use the chicken language by adding a ‘p’ letter before and after a vowel.

Simulate the new game and display the results.

ANSWER: 


#include <stdio.h>
#include <string.h>

int isVowel(char a)
{
  if (a == 'a' || a == 'e' || a == 'i' || a == 'o' || a == 'u' || a == 'A' || a == 'E' || a == 'I' || a == 'O' || a == 'U')
  {
    return 1;
  }
  else
    return 0;
}

int main()
{
  // defining varibales for storing the string
  char string[100];
  char chicken_string[100];
  char continue_game = 'y';
  int i = 0, j = 0;

  while (continue_game == 'y' || continue_game == 'Y')
  {
    printf("Enter a string to be converted to chicken language:\t");
    scanf("%s", string);
    while (string[i] != '\0')
    {
      if (isVowel(string[i]))
      {
        chicken_string[j] = 'p';
        j++;
      }

      chicken_string[j] = string[i];
      j++;

      if (isVowel(string[i]))
      {
        chicken_string[j] = 'p';
        j++;
      }
      i++;
    }

    chicken_string[j] = '\0';

    printf("The chicken string is :\t%s\n", chicken_string);
    printf("Would you like to continue---\nTo continue press y\n");
    scanf("%s", continue_game);
  }
}
