        public static String decimalTobinary(String decimal) {
            int num = Integer.parseInt(decimal);
            String binary = "";
            while(num > 0){
                binary = (num & 1) + binary;
                num = num >> 1;
            }
            return binary;
        }

        public static String binaryTodecimal(String binary) {
            int num = 0;
            int power2 = 1;
            for(int i = binary.length()-1; i>= 0; i--){ //start from right most end
                if(binary.charAt(i) == '1')
                    num += power2;
                power2 = power2 * 2;//get the next power of 2
            }
            return "" + num;
