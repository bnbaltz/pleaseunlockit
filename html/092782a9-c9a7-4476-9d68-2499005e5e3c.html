# method that applies diagonal sort
def diagonalSort(a):
    # get the length of list into n
    n=len(a)
    
    # start from k=0 to n-1
    for k in range(n):
        # initialise i to k and j to 0, which means start traversing diaginally from first column
        i=k
        j=0
        
        # declare an empty temporary list
        temp_list=[]
        
        # traverse list diagonally till i>=0
        while i>=0:
            # append a[i][j] values into temo_list
            temp_list.append(a[i][j])
            
            # decrement i by 1
            i-=1
            
            # increment j by 1
            j+=1
            
        # sort the temp_list
        temp_list.sort()
        
        # using the same logic which we have used above, replace elements in diagonal with sorted temp_list values
        i=k
        j=0
        index=0
        while i>=0:
            a[i][j]=temp_list[index]
            i-=1
            j+=1
            index+=1
            
    # start from k=1 to n-1
    for k in range(1,n,1):
        # initialise i to n-1 and j to 1, which means start traversing diaginally from last row
        i=n-1
        j=k
        
        # declare an empty temporary list
        temp_list=[]
        
        # traverse list diagonally till i>0 and j<n
        while i>0 and j<n:
            # append a[i][j] values into temo_list
            temp_list.append(a[i][j])
            
            # decrement i by 1
            i-=1
            
            # increment j by 1
            j+=1
            
        # sort the temp_list
        temp_list.sort()
        
        # using the same logic which we have used above, replace elements in diagonal with sorted temp_list values
        i=n-1
        j=k
        index=0
        while i>0 and j<n:
            a[i][j]=temp_list[index]
            i-=1
            j+=1
            index+=1
