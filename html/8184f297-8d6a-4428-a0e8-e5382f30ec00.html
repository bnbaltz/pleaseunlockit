#include <stdio.h>
#include <stdlib.h>
_Bool IsDecimalInteger(char str[]){
	//initialize i with 0 to traverse the  string str
	int i=0;
	//if the string starts with - sign, it could be a negative number so we move forward
	if(str[0]=='-'){
		i++;
	}
	//if the string ends without any digit, then this is not an integer so we return 0
	if(str[0]=='\0')
		return 0;
	//we loop through the string and check each character for a digit character, if not we return 0
	while(str[i]!='\0'){
		if(str[i]<'0'||str[i]>'9'){
			return 0;
		}
		i++;
	}
	//if our program reaches here, it means that we have checked all the characters and they are all digits
	//we can now return 1
	return 1;
}

char GetSortingOrder(double arr[], size_t arr_size){
	//if size is 0 it is an error situation
	if(arr_size==0)
		return -1;
	//if size is 1 then by default it is a constant array
	if(arr_size==1)
		return 3;
	// lets assume that this is is constant and sorted and reverse sorted
	// So that we can loop through the array and check if each of them is valid or not
	int is_constant=1;
	int is_sorted=1;
	int is_reverse_sorted=1;
	//loop through the array
	for(int i=1;i<arr_size;i++){
		//if we have already identified that this is not a constant array then dont check
		//if it is a constant array then all elements must be equal, if not then we know that it is not constant
		if(is_constant && arr[i]!=arr[i-1]){
			is_constant=0;
		}
		//if we have already identified that this is not a sorted array then dont check
		//if it is a sorted array then all elements must be greater than or equal to their preceding element, if not then we know that it is not sorted
		if(is_sorted && arr[i]<arr[i-1]){
			is_sorted=0;
		}
		//if we have already identified that this is not a reverse sorted array then dont check
		//if it is a reverse sorted array then all elements must be less than or equal to their preceding element, if not then we know that it is not reverse sorted
		if(is_reverse_sorted && arr[i]>arr[i-1]){
			is_reverse_sorted=0;
		}
		//if we know that this array is neither sorted, nor constant nor reverse sorted, then no need to check further
		if(!is_constant && !is_sorted && !is_reverse_sorted)
			break;
	}
	//check if constant
	if(is_constant){
		return 3;
	}else if(is_sorted){//check if sorted
		return 1;
	}else if(is_reverse_sorted){//check if reverse sorted
		return 2;
	}else{//else it is unsorted
		return 0;
	}
}

void RemoveExtraSpaces(char txt[]){
	int i, x;
  	for(i=x=0; txt[i]; i++)
	//add next character only if it is not a space or if the preceding character is not a space
    if(txt[i] != ' ' || (i > 0 && txt[i-1] != ' '))
      txt[x++] = txt[i];
	//there could be a space left at the end
	if(x-1>0&&txt[x-1]==' '){
		txt[x-1]='\0';
	}else{
		txt[x] = '\0';
	}
}
