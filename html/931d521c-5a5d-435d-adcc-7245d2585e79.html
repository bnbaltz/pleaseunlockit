import java.util.Scanner;

public class Test {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        //taking user input
        int t = sc.nextInt();
        sc.nextLine();
        String[] data = new String[t];
        for (int i = 0; i < t; i++) {
            data[i] = sc.nextLine();
        }
        sc.close();

        System.out.println();
        int max, min, sum;
        for (int i = 0; i < t; i++) {
            //converting string data to int
            String[] ar = data[i].split(" ");
            int a = Integer.parseInt(ar[0]);
            int b = Integer.parseInt(ar[1]);

            //finding min and max value
            max = (a > b) ? a : b;
            min = (a < b) ? a : b;

            //getting the sum
            sum = getSumRange(min, max);
            System.out.println(sum);//printing the sum

        }

    }
    //for finding the sum in the range 
    private static int getSumRange(int min, int max) {
        int sum = 0;
        
        //calculating the sum 
        for (int i = min; i <= max; i++) {
            sum += i;
        }
        return sum;
    }
}
