public int changeAtIndex(String word, int index, char letter){
    try{
        word = word.substring(0,index)+letter+word.substring(index+1);
    }
    catch (StringIndexOutOfBoundsException s){
        return 1;
    }
    catch (ArrayIndexOutOfBoundsException a){
        return -1;
    }
    return 0;
}
