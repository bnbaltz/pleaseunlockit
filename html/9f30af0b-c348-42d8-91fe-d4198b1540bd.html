for i= 1:n
    x(i) = input('');
end

%applying selection sort algorithm
for i=0:n-1
    index = 1
    for j=2:n-i   %loop to find index of max value
        if(x(j) > x(j-1))
            index = j
        end
    end
    %swap the value from the index to its position
    y = x(index)
    x(index) = x(n-i)
    x(n-i) = y
end
%display sorted array
disp(x)
