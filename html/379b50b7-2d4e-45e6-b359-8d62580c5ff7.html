#include<iostream>
#include<fstream>
using namespace std;

//Here we are searching target in sorted arr[] using Linear search Algorithm
int linearSearch(int arr[], int size, int target){
    for(int i=0;i<size;i++){
        cout<<i<<" ";
        if(arr[i] == target) break;
    }
}

//Here we are searching target in sorted arr[] using Binary search Algorithm
void binarySearch(int arr[], int size, int target) {
    int left = 0, right = size-1;
    while (left <= right) {
        int mid = (left + right) / 2; //finding mid index value
          cout<<mid << " ";
        // Check if target is present at index mid
        if (arr[mid] == target) break;
 
        // If target is greater, search in right half
        if (arr[mid] < target)
            left = mid + 1;
 
        // If target is smaller, search in left half
        else
            right = mid - 1;
    }
   
}
