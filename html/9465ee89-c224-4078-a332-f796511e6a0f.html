#include <stdio.h>
#include <string.h>


void fun(char *arr,int len){

	int i=0;
	int j=0;

	// remove all leadins spaces.
	while(j<len && arr[j]==' '){
		j++;
	}

	//printf("dong for %d\n",i);
	//printf("dong for %d\n",j);

	// remove extra spaces from rest of string
	while(j<len){

		if(arr[j]==' '){

			arr[i]=arr[j];
			i++;j++;

			while(j<len && arr[j]==' '){
				j++;
			}

		}

		else if(arr[j]=='.' || arr[j]==',' || arr[j]=='?' ){
			/* In my code I put one space by default , we need to remove that default space.
			   i.e. "geeks    ." will be give "geeks ."  if we do not include below if statement
			*/
			if(arr[i-1]==' '){
				arr[i-1]=arr[j];
				j++;
			}
			else{
				arr[i]=arr[j];
				i++;j++;
			}

		}
		else{

			arr[i]=arr[j];
			i++;j++;
		}

		//printf("dong for %d\n",i);
		//printf("dong for %d\n",j);

	}

	/* In my above code, iff the string  is like "Geek.  " , then I put one space(" ") by default.
	   we need to remove this space at the end of string
	*/
	if(i<j){

		i--;
		if(arr[i]==' '){
			arr[i]='\0';
			i++;
		}
		while(i<len){
			arr[i]='\0';
			i++;
		}

	}
}

int main(void){

	char str[]=" Hello,   Students  !    ";
	printf("The input text is\n");
	printf("%s\n",str);
	int len=strlen(str);

	fun(str,len);

	printf("\nThe modified text is\n");
	printf("%s\n",str);

}
