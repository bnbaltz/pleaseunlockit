#include<iostream>
using namespace std;
int main(){
   double Base_price[]={0,100,130,150,160,165};/*Base Price of Subjets in portraits*/
   int no_of_sub;                              /*store no of subject in portrait choosen by user*/
   char fancy_background,appointment_date;     /*to store fancy background and appointment date choosen or not*/
   double total_price;                         /*To store total price*/
   cout<<"Enter the number of subjects in the portrait: ";
   cin>>no_of_sub;
   cout<<"Do you want a fancy background (y/n)? ";
   cin>>fancy_background;
   cout<<"Do you want an appointment date (y/n)? ";
   cin>>appointment_date;
   if(no_of_sub>=5){                          /*if the choosen subject no is 5 or more*/
       total_price=Base_price[5];
       if(fancy_background == 'y'&&appointment_date == 'Y'){          /*checking if fancy background and appointment date both choosen or not*/
           total_price+=2*(Base_price[5]/10);
       }
       else if(fancy_background == 'y'||appointment_date == 'y'){    /*checking if fancy background or appointment date not both choosen or not*/
           total_price+=(Base_price[5]/10);
       }
   }
   else{
       total_price=Base_price[no_of_sub];
       if(fancy_background == 'y'&&appointment_date == 'y'){
           total_price+=2*(Base_price[no_of_sub]/10);
       }
       else if(fancy_background == 'y'||appointment_date == 'y'){
           total_price+=(Base_price[no_of_sub]/10);
       }

   }
   cout<<"The price is: $"<<total_price;
   


}
