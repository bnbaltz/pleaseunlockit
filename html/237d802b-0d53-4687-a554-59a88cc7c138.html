using System;

 

class Program {

static void Main(string[] args) {

// Variable declaration

int numA = 0, numB = 0, numC = 0, numD = 0, numF = 0, i = 0;

int numStudents;

char letterGrade;

// Take number of students as input

Console.Write("Enter the number of students: ");

numStudents = Convert.ToInt32(Console.ReadLine());

 

// Declare array of chars

char []letterGrades = new char[numStudents];

 

// Do-while loop

do {

// Take a letter grade as input from the user

Console.Write("Enter the letter grade of a student(Q to stop): ");

letterGrade = Char.ToLower(Console.ReadLine()[0]);

 

// Switch-case to keep track of number of each grade entered

switch(letterGrade) {

case 'a':

numA++;

break;

case 'b':

numB++;

break;

case 'c':

numC++;

break;

case 'd':

numD++;

break;

case 'f':

numF++;

break;

}

i++;

} while (i < numStudents && letterGrade != 'Q' && letterGrade != 'q');

 

// Display the total number of each letter grade entered

Console.WriteLine("\nNumber of each letter grades:");

Console.WriteLine("A: {0}", numA);

Console.WriteLine("B: {0}", numB);

Console.WriteLine("C: {0}", numC);

Console.WriteLine("D: {0}", numD);

Console.WriteLine("F: {0}", numF);

}

}
