QUESTION:

Code the following in C...

•Hard-code a string of your choice. Print out your string.

•Generate a random key of chars the same length as your string. Print out the key.

•Exclusive OR (XOR) each character in your string with the corresponding character in the random key to generate the ciphertext. Print the ciphertext.

•Exclusive OR (XOR) each character in the ciphertext with the corresponding character in the random key to generate the original text. Print out the original text.

•Watch out for the impact of the XOR and random generation on null string terminators.

ANSWER:


ANSWER:-

CODE:-

#include <stdio.h>
#include <string.h>
int main(){
   char cipher[30],key[30], decrypt[30];
   char plain[] = "This is C Programming.";
   printf("Plain text: %s\n",plain);
   int i,len = strlen(plain);
   // Key generation
   srand(time(0));
   for(i=0;i<len;i++){
       key[i] = (char)(rand()%26 + 97);
   }
   key[i] = '\0';
   printf("Key: %s\n",key);
   // xor key with plain text
   for(i=0;i<len;i++){
       cipher[i] = (char)(plain[i]^key[i]);
   }
   cipher[i] = '\0';
   for(i=0;i<len;i++){
       printf("%02X",cipher[i]);
   }
   // xor cipher text with cipher text
   for(i=0;i<len;i++){
       decrypt[i] = (char)(cipher[i]^key[i]);
   }
   decrypt[i] = '\0';
   printf("\nDecrypted Plain text: %s\n",decrypt);
}
