import java.util.*;


public class Sound {

public static void main(String[] args) {
  
int[] samples1 = new int[]{40, 2532, 17, -2300, -17, -4000, 2000, 1048, -420, 33, 15, -32, 2030, 3223};
  
Sound s1 = new Sound(samples1);
System.out.println("The original array is " + Arrays.toString(s1.getSamples() ) );
System.out.println( s1.limitAmplitude(2000) + " sounds had to limit amplitude" );
System.out.println( "Updated array is " + Arrays.toString( s1.getSamples() ) );
  
System.out.println("");
  
int[] samples2 = new int[]{0, 0, 0 , 0 -14, 0, -35, -39, 0, -7, 16, 32, 37, 29, 0, 0};
  
Sound s2 = new Sound(samples2);
System.out.println("Original Array: " + Arrays.toString( s2.getSamples() ) );
s2.trimSilenceFromBeginning();
System.out.println("Updated Array: " + Arrays.toString( s2.getSamples() ) );
 
  
}

private int[] samples;
  
public int[] getSamples() {
return samples;
}
  
public Sound(int[] samples) {
this.samples = samples;
}
  

public int limitAmplitude(int limit)
{
  int cnt=0;
  for(int i=0;i<samples.length;i++)
    {
        //check the given no is greater than limit if yes then set limit value
        if(samples[i] >= limit)
        {
            samples[i] = limit;
            cnt++;
        }
        //assign negative limit value to temp variable and check negative limit value
        int temp= -limit;
        if(samples[i] <= temp )
        {
            samples[i] = temp;
            cnt++;
        }
    }
    //return count of total exceed value of limit
    return cnt;
  
}
  
public void trimSilenceFromBeginning()
{
    //cnt is use to count the starting zeros of array
    int cnt=0;
  for(int i=0;i<samples.length;i++)
    {
        if(samples[i]!=0)
        {
            break;
        }
        else
        {
            cnt++;
        }
    }
    //when loop break we get the total count of starting zeros
    //create new array of total length - count size
    int[] temp = new int[samples.length - cnt];
    //loop start with cnt and assign value to temp array
    for(int i=cnt,j=0;i<samples.length;i++,j++)
    {
        temp[j] = samples[i];
    }
    //then recreate a sample array of size temp and assign all value of temp to samples
    samples = new int[temp.length];
    samples = temp;
}

}
