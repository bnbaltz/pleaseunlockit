(I used (char)254 for the piece, you can use whatever you want to represent it.)

 

The method should then prompt the user to make any number of moves using N, S, E, W and update the position of the piece in the 2d array. N will move the piece up on the board, S down, E to the right, and W to the left.

 

If the piece would go off of the board, it should appear on the other side of the board as if it was ‘wrapping’ around the board. So if the piece was at the top and the user types N to go up more, it should appear at the bottom in the same column.

 

You do not need to print out the board between moves. The method ends when the user enters Quit, and the board with the new piece location is then printed.

import java.util.Scanner;

public class solution {

    static void display(int arr[][]) {

        for (int i = 0; i < arr.length; i++) {
            for (int j = 0; j < arr[0].length; j++) {
                if (arr[i][j] == 1)
                    System.out.print("*");
                else
                    System.out.print("-");
            }
            System.out.println();
        }
    }
    //It will fill 1 where piece is present in all other location it will be 0
    static void changePostion(int board[][], int positionX, int positionY) {

        for (int i = 0; i < 9; i++) {
            for (int j = 0; j < 9; j++) {

                if (i + 1 == positionX && j + 1 == positionY)
                    board[i][j] = 1;
                else
                    board[i][j] = 0;
            }
        }

    }

    public static void main(String[] args) {
        Scanner scnr = new Scanner(System.in);

        int board[][] = new int[9][9];

        int positionX = 5, positionY = 5;

        changePostion(board, positionX, positionY);
        display(board);

        while (true) {

            System.out.println("N.Up\nS.Down\nW.Left\nE.righ\nQ.Quite");

            String choice = scnr.next();
            boolean exit = false;

            switch (choice) {

            case "N":
                positionX = (positionX - 1);
                if (positionX <= 0)
                    positionX = positionX + 9;

                break;
            case "S":
                positionX = (positionX + 1);
                if (positionX > 9)
                    positionX = positionX - 9;

                break;
            case "W":
                positionY = (positionY - 1);
                if (positionY <= 0)
                    positionY = positionY + 9;

                break;
            case "E":
                positionY = (positionY + 1);
                if (positionY > 9)
                    positionY = positionY - 9;

                break;
            case "Q":
                exit = true;
                break;
            default:
                System.out.println("Enter Valid Choice");
                break;

            }
            if (exit)
                break;
            changePostion(board, positionX, positionY);
            //If you want to display at each move then uncomment this display(board)
            // display(board);
        }

        System.out.println("Current position Of piece:" + positionX + " , " + positionY);

    }
