'use strict';

document.addEventListener("DOMContentLoaded", function(){
    var meta = document.createElement("META");
    meta.setAttribute("charset", "utf-8");
    document.head.appendChild(meta);

    var title = document.createElement("TITLE");
      var t = document.createTextNode("Fix the Page");
      title.appendChild(t);
      document.head.appendChild(title);

      var link = document.createElement("LINK");
      link.setAttribute("rel", "stylesheet");
      link.setAttribute("type", "text/css");
      link.setAttribute("href", "styles.css");
      document.head.appendChild(link);

    var body = document.createElement("BODY");

    // h1 header start
    var h1 = document.createElement("H1");
    h1.setAttribute("id", "main-title");
    t = document.createTextNode("Fix ");
    h1.appendChild(t);
    var em = document.createElement("EM");
    t = document.createTextNode("this ");
    em.appendChild(t);
    h1.appendChild(em);
    t = document.createTextNode("Page");
    h1.appendChild(t);   
    document.body.appendChild(h1);
    // h1 header end

    // img start
    var img = document.createElement("IMG");
      img.setAttribute("src", "mars.jpg");
      img.setAttribute("width", "500");
      img.setAttribute("height", "500");
      img.setAttribute("alt", "images of mars");
      document.body.appendChild(img);
      // img end

    // knowledge lists ol / li start
    var engageLists = ["Ready", "Set", "Go!"];
    var ol = document.createElement("OL");
    ol.setAttribute("id","engage");
    document.body.appendChild(ol);
    engageLists.forEach(function(engageList) {
        var li = document.createElement("LI");
        t = document.createTextNode(engageList);
        li.appendChild(t);
        document.getElementById("engage").appendChild(li);
    }); 
    // knowledge lists ol / li end

    // h3 header start
    var h3 = document.createElement("H3");
    t = document.createTextNode("List of things we need to learn about JavaScript");
    h3.appendChild(t);   
    document.body.appendChild(h3);
    // h3 header end

    // knowledge lists ul / li start
    var knowledgeLists = ["Relation to HTML", "Variables", "Syntax", "Conditionals", "Loops", "Functions", "Using the DOM"];
    var ul = document.createElement("UL");
    ul.setAttribute("id","knowledge-list");
    document.body.appendChild(ul);
    knowledgeLists.forEach(function(knowledgeList) {
        var li = document.createElement("LI");
        t = document.createTextNode(knowledgeList);
        li.appendChild(t);
        document.getElementById("knowledge-list").appendChild(li);
    }); 
    // knowledge lists ul / li end

    // h6 header start
    var h6 = document.createElement("H6");
    h6.setAttribute("class","change-me");
    t = document.createTextNode("For extra credit change this tag to an h6 tag.");
    h6.appendChild(t);   
    document.body.appendChild(h6);
    // h6 header end

    // asside start
    var asside = document.createElement("ASIDE");
      document.body.appendChild(asside);
      var para = document.createTextNode("JavaScript is only fun when we can use it to manipulate HTML!");
      asside.appendChild(para);
      // asside end
    
})
