class Birthday():

    def __init__(self,day,month,year):
        self.birth_day=day
        self.birth_month=month
        self.birth_year=year

    def __str__(self):
        return str(self.birth_month)+'/'+str(self.birth_day)+'/'+str(self.birth_year)

    def __hash__(self):
        return (self.birth_month+self.birth_day+self.birth_year)%12

    def __eq__(self, other):
        return self.birth_year==other.birth_year and \
            self.birth_day==other.birth_day and \
            self.birth_month==other.birth_month


def main():
    hashtable=[]
    for i in range(12):
        hashtable.append([])
    with open('F:\\bdaylist.txt','r') as read_file:
        for line in read_file.readlines():
            line=line.strip().split()
            b = Birthday(int(line[0]),int(line[1]),int(line[2]))
            hashtable[b.__hash__()].append(b)
    for i in range(12):
        print('Hash location',i,'has',len(hashtable[i]),'elements in it')
main()
