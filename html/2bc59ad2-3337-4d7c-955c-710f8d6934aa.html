import random


def generateRandomList():
    result = []
    for i in range(500):
        result.append(random.randint(-500, 700))
    return result


def printList(arr):
    print "List : "
    for x in arr:
        print (x)


random_list = generateRandomList()
print "Max : ", max(random_list)
print "Max : ", min(random_list)
print "Sum : ", sum(random_list)
random_list.sort(key=lambda x: -x)
printList(random_list)
