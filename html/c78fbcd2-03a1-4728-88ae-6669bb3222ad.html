// C# program to find the factorial of an integer

using System;

class Factorial {
    
    // method to return the factorial of n
    static int factorial(int n)
    {
        int result = 1; // initialize result to 1
        // loop from n to 1 multiplying i to result
        for(int i=n;i>=1;i--)
            result = result*i;
            
        return result;    
    }
    
  static void Main() {
        
        int n;
        // input n
        Console.Write("Enter an integer: ");
        n = Convert.ToInt32(Console.ReadLine());
        
        // get the factorial of n
        int result = factorial(n);
        // display result
        Console.WriteLine("The Factorial of "+n+" is: "+result);
  }
}

// end of program
