void printList(struct node *head){
    struct node *temp=head;
    while(temp->next!=NULL){
        printf("%s %s ",temp->name,temp->surname);
        if(temp->sibling==NULL){
            printf("No siblings for this person");
        }
        else{
            printf("Sibling name of this person is %s",temp->sibling->name);
        }
        temp=temp->next;
    }
}
void freeNode(struct node *node){
    node->next=NULL;
    node->sibling=NULL;
    free(node);
}
struct node * siblings(struct node **head){
    struct node *temp=head;
    struct node *ttemp=NULL;
    int count=0,tcount=0;
    while(temp->next!=NULL){
        count=0;
        if(temp->sibling!=NULL){
            ttemp=head;
            tcount=0;
            while(ttemp->next !=NULL && strcmp(ttemp->surname,temp->surname)==0 && count==tcount){
                tcount++;
                ttemp=ttemp->next;
            }
            if(strcmp(ttemp->surname,temp->surname)==0 || ttemp->next!=NULL)
                temp->sibling=ttemp;        
        }
        count++;
        temp=temp->next;
    }
    return head;
}
void insert(struct node **head,char name[],char surname[]){
    struct node* temp=NULL;
    struct node* prev=NULL;
    struct node* nnode = (struct node*)malloc(sizeof(struct node)); 
    nnode->name=name;
    nnode->surname=surname;
    nnode->next=NULL;
    nnode->sibling=NULL;
    if(head==NULL){
        nnode->next=(head);
        (head)=nnode;
    }
    else if(head!=NULL){
        prev=temp;
        temp=head;
        while((strcmp(nnode->name,temp->name))>0 && (temp->next!=NULL)){
            prev=temp;
            temp=temp->next;
        }
        if(strcmp(nnode->name,temp->name)==0){
            if(strcmp(nnode->surname,temp->surname)<0){
                nnode->next=temp;
                prev->next=nnode;
            }
            else{
                nnode->next=temp->next;
                temp->next=nnode;
            }
        }
        if(temp->next!=NULL){
            nnode->next=temp;
            prev->next=nnode;
        }
        else if(temp->next==NULL){
            temp->next=nnode;
            nnode->next=NULL;
        }
    }
    //for siblings mathing in the 2d array
    head=siblings(&head);
}
struct node *delete(struct node **head,char name[],char surname[]){
    struct node *prev=NULL;
    struct node * temp=NULL;
    if(head==NULL){
        return NULL;
    }
    prev=temp;
    temp=head;
    while(temp->next!=NULL){
        if(strcmp(temp->name,name)==0 && strcmp(temp->surname,surname)==0){
            if(temp==head){
                head=temp->next;
                freeNode(temp);
                return head;
            }
            else if(temp->next==NULL){
                prev->next=NULL;
                freeNode(temp);
                return head;
            }
            else if(temp!=head && temp->next!=NULL){
                prev->next=temp->next;
                freeNode(temp);
                return head;
            }
            prev=temp;
            temp=temp->next;
        }
