int validate(char *s1) {
    int n = strlen(s1);
    int count = 0;
    
    while(*s1!='\0' && count < n) {
        char ch = *s1;
        
        // check if last 4 characters of string equals .edu
        if(count == n-4 && ch != '.')
            return 0;
        else  if(count == n-3 && ch != 'e')
            return 0;   
        else if(count == n-2 && ch != 'd')
            return 0;    
        if(count == n-1 && ch != 'u')
            return 0;
            
        *s1++;
        count++;
    }
    return 1;    
}
