#include<iostream>
#include<vector>
using namespace std;
int main()
{
    vector<int> A;
    vector<int> res;
    int num,i;
    cin>>num;                 // Taking size of array as input
    for(i=0;i<num;i++)        // Taking input for elements of array
    {
        int temp;
        cin>>temp;
        A.push_back(temp);
    }
    res.push_back(0);         // Their is no element before A[0]
    for(i=1;i<num;i++)
    {
        int counter=0;
        for(int j=0;j<i;j++)   // Checking for all elements before the current index
        {
            if(A[j]>=A[i])
                counter++;     // Increasing counter if any number greater than current number
        }
        res.push_back(counter); // Pushing it into the array
    }
    for(i=0;i<res.size();i++)
    {
        cout<<res[i]<<" ";
    }
    return 0;
}
