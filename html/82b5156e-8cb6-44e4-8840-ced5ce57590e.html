IDedLinkedList.java

import java.util.LinkedList;

/**IdedLinkedList contains generic type of elements*/
public class IDedLinkedList <T extends IDedObject>{

   private LinkedList<T> list;

   /**default constructor*/
   public IDedLinkedList() {
       super();
       list = new LinkedList<T>();
   }

   /**parameterized constructor*/
   public IDedLinkedList(LinkedList<T> list) {
       super();
       this.list = list;
   }
  
   /**to empty the list*/
   public void makeEmpty() {
       list.clear();
   }
  
   /**to find the item using its id*/
   public T findId(int id) {
  
       for(T item : list)
       {
           if(item.getID() == id)
           {
               return item;
           }
       }
       return null;
   }
  
   /**to insert item in the list at begining */
   public boolean insertAtFront(T item) {
       list.addFirst(item);
       return true;
   }
  
   /**to delete the first item*/
   public T deleteFromFront() {
       return list.pollFirst();
   }
  
   /**to delete the item using its id*/
   public T delete(int id) {
       for(int i = 0 ; i< list.size();i++)
       {
           if(list.get(i).getID() == id)
           {
               return list.get(i);
           }
       }
       return null;
   }
  
   /**to print the sum of ids*/
   public int printTotal() {
      
       if(list.isEmpty())
           return -1;
       int sum = 0;
       for(T item : list)
       {
           sum = sum + item.getID();
       }
       return sum;
   }
  
}

import java.util.LinkedList;
import java.util.List;

/**MyItem contains the infromation about the item*/
public class MyItem implements IDedObject {

   private int itemId;
   private int itemPrice;
   private List<Integer> itemDescription;
  
   /**default constructor*/
   public MyItem() {
       super();
       itemDescription = new LinkedList<Integer>();
   }

   /**parameterized constructor*/
   public MyItem(int itemId, int itemPrice, List<Integer> itemDescription) {
       super();
       this.itemId = itemId;
       this.itemPrice = itemPrice;
       this.itemDescription = itemDescription;
   }


   /**to get the id of the item*/
   public int getItemId() {
       return itemId;
   }
  
   /**to set the id of the item*/
   public void setItemId(int itemId) {
       this.itemId = itemId;
   }

   /**to get the price of the item*/
   public int getItemPrice() {
       return itemPrice;
   }

   /**to set the price of the item*/
   public void setItemPrice(int itemPrice) {
       this.itemPrice = itemPrice;
   }


   public List<Integer> getItemDescription() {
       return itemDescription;
   }


   public void setItemDescription(List<Integer> itemDescription) {
       this.itemDescription = itemDescription;
   }

  
   /**overriding getID method of the MyItem*/
   @Override
   public int getID() {
  
       return this.itemId;
   }

   /**Print the details of the item in one line*/
   @Override
   public String printID() {
       StringBuilder details = new StringBuilder();
       details.append(this.itemId);
       details.append(" "+this.itemPrice);
       for(int i : itemDescription)
           details.append(" "+i);
       return details.toString();
   }   
}

import java.io.BufferedReader;
import java.io.BufferedWriter;
import java.io.FileNotFoundException;
import java.io.FileReader;
import java.io.FileWriter;
import java.io.IOException;
import java.util.LinkedList;
import java.util.List;

/**P2Driver to initialize the program */
public class P2Driver {

   private static IDedLinkedList<MyItem> list = new IDedLinkedList<MyItem>();
   private static FileReader input;
   private static FileWriter output1;
   private static BufferedWriter output;
  
   /**to get the item and print it to output file*/
   private static void findId(int id) throws IOException
       {
           MyItem item = list.findId(id);
           if(null == item)
           {
               output.write("Null");
           }
           else
           {
               output.write(item.printID()+"\n");
           }
           output.append(System.lineSeparator());
       }
  
   /**insert the item in the list and prints it status to output file*/
   private static void insert(String[] inputs) throws IOException {
       if(inputs.length < 3)
       {
           output.write("error in Input");
           output.append(System.lineSeparator());
       }
       int id = Integer.valueOf(inputs[1]);
       int price = Integer.valueOf(inputs[2]);
       List<Integer> descriptionList = new LinkedList<Integer>();
       for(int i=3; i<inputs.length; i++)
       {
           descriptionList.add(Integer.valueOf(inputs[i]));
       }
       MyItem item = new MyItem(id, price, descriptionList);
       boolean check = list.insertAtFront(item);
       output.write(check+"");
       output.append(System.lineSeparator());
   }
  
   /**to delete the item using id and prints its details */
   private static void deleteId(int id) throws IOException {
       MyItem item = list.delete(id);
       output.write(item.printID()+"");
       output.append(System.lineSeparator());
   }
  
   /**to delete the first item in the list*/
   private static void delete() throws IOException {
       MyItem item = list.deleteFromFront();
       output.write(item.printID()+"");
       output.append(System.lineSeparator());
   }
  
   /**to split the line of the input file and calls corresponding function*/
   private static void executeCommand(String command) throws NumberFormatException, IOException {
       String[] inputs = command.split(" ");
       switch(inputs[0])
       {
       case "FindID":
       {
           findId(Integer.valueOf(inputs[1]));
       }
       break;
       case "Insert":
       {
           insert(inputs);
       }
       break;
       case "DeleteID":
           deleteId(Integer.valueOf(inputs[1]));
       break;
       case "Delete":
           delete();
       break;
       case "PrintTotal":
       {
           output.write(list.printTotal()+"");
           output.append(System.lineSeparator());
       }
       break;
       default:
           output.append("Error");
           output.append(System.lineSeparator());
       }
   }
  
   /**main method to initialize the program*/
   public static void main(String[] args) {
      
       if(args.length !=2)
       {
           System.out.println("File name not provided");
           return;
       }
      
       try {
           input = new FileReader(args[0]);
           BufferedReader br = new BufferedReader(input);
           output1 = new FileWriter(args[1]);
       output = new BufferedWriter(output1);
           String command;
           while((command = br.readLine()) != null)
           {
               if( command.isEmpty() || '#' == command.charAt(0))
                   continue;
               if(command.split(" ")[0].equals("End"))
                   break;
               executeCommand(command);
           }
           input.close();
           output.close();
           } catch (FileNotFoundException e) {
               System.out.println("File not found!!");
               return;
           } catch (IOException e) {
               e.printStackTrace();
               return;
           }
}
}
