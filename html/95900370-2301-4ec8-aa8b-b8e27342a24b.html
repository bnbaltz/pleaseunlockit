import java.io.*;
import java.util.Scanner;
public class NoonSnooze
{
    public static void main(String[] args)
    {
        int snooze;
        Scanner scan=new Scanner(System.in);
        snooze=scan.nextInt();       //read integer snooze
        int hh;
        int mm;
        mm=snooze%60;        //find extra minutes those are not form an hour
        hh=snooze-mm;        //substract extra minutes
        hh=hh/60;           //divide remaining minutes by 60 for hours
        hh=hh+12;           //add 12 hours for count starts from 12.00pm
        hh=hh%24;           //get 24 hours clock for decide am or pm
        String a="am";
        if(hh>=12)          //if hours >= 12 it is pm else am
        {
            a="pm";
        }
        hh=hh%12;           // convert to 12 hours clock for display
        if(hh==0)           // if hours = 0 take as 12 hours
        {
            hh=12;
        }
        System.out.print(hh+":"+mm+a); //print clock
    }
}
