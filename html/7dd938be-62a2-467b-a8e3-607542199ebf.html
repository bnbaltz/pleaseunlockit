# to enter the details of 3 students.

    # Prompt the user to enter the second score.

    # Store the second score in the variable grade2.

    grade2=int(input("Enter the score 2: "))

 

    # Prompt the user to enter the third score.

    # Calculate the average of the three grades.

    # Store the average in the variable average.

    average=(grade1+grade2+grade3)/3
