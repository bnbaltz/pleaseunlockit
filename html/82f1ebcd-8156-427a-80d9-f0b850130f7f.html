package chegg;

import java.util.ArrayList;
import java.util.List;

// Student structure with name and quizScore array 
class Student {
	static final int SIZE = 5;

	String name;
	int quizScore[] = new int[SIZE];

	public Student(String name, int[] quizScore) {
		super();
		this.name = name;
		this.quizScore = quizScore;
	}

	// get specific quiz Score - 0 based index
	int getQuizScore(int number) {
		return quizScore[number];
	}

	// set specific quiz Score - 0 based index
	void setQuizScore(int number, int newScore) {
		quizScore[number] = newScore;
	}

	// get student name
	String getName() {
		return name;
	}

	// set student name
	void setName(String newName) {
		name = newName;
	}

	// to print student data in proper format
	@Override
	public String toString() {
		System.out.printf("%-10s%8d%15d%16d%17d%16d\n", name, quizScore[0], quizScore[1], quizScore[2], quizScore[3],
				quizScore[4]);

		return "";
	}

}

//Test student structure
public class Client {
	static final int SIZE = Student.SIZE;

	// print student details in proper format
	static void printStudentsDetails(List<Student> students) {
		System.out.println("Name\t\tQ1\t\tQ2\t\tQ3\t\tQ4\t\tQ5");
		System.out.println("----------------------------------------------------------------------------------");
		for (Student student : students) {
			student.toString();
		}
		System.out.println();
	}

	// replace specific student name
	static void repalceStudentName(List<Student> students, String oldName, String newName) {
		for (Student student : students) {
			if (student.getName().equals(oldName)) { // find student in students array list
				student.setName(newName); // when found change name to newName
				break; // break the loop
			}
		}
		printStudentsDetails(students);

	}

	// replace specific student's score with new score
	static void repalceStudentScore(List<Student> students, String name, int number, int newScore) {
		for (Student student : students) {
			if (student.getName().equals(name)) {
				student.setQuizScore(number, newScore);
				break;
			}
		}
		printStudentsDetails(students);
	}

	// replace whole student data
	static void repalceStudent(List<Student> students, String oldName, String newName, int[] score) {
		for (Student student : students) {
			if (student.getName().equals(oldName)) {
				student.setName(newName);
				for (int i = 0; i < SIZE; i++)
					student.setQuizScore(i, score[i]);
				break;
			}
		}
		printStudentsDetails(students);
	}

	// add student data before specific student by name
	static void addStudentBeforeAnother(List<Student> students, String oldName, String newName, int[] score) {
		int index = 0;
		for (Student student : students) {
			if (student.getName().equals(oldName))
				break;

			index++;
		}
		students.add(index, new Student(newName, score));

		printStudentsDetails(students);
	}

	// delete student by name
	static void deleteStudent(List<Student> students, String name) {
		int index = 0;
		for (Student student : students) {
			if (student.getName().equals(name))
				break;

			index++;
		}
		students.remove(index);

		printStudentsDetails(students);
	}

	public static void main(String[] args) {
		List<Student> students = new ArrayList<>();
		students.add(new Student("name1", new int[] { 1, 2, 3, 4, 5 }));
		System.out.println("Pre-Populated data:");
		printStudentsDetails(students);

		System.out.println("Replace student name name1 to name2:");
		repalceStudentName(students, "name1", "name2");

		System.out.println("Replace student score of 1st quiz of name2 to 10:");
		repalceStudentScore(students, "name2", 0, 10);

		System.out.println("Replace student with name2 as name3 with scores {25,43,23,2,1}:");
		repalceStudent(students, "name2", "name3", new int[] { 25, 43, 23, 2, 1 });

		students.add(new Student("name1", new int[] { 1, 2, 3, 4, 5 }));
		System.out.println("Adding a student at last:");
		printStudentsDetails(students);

		System.out.println("Add student with name4 bewfore name3 with scores {2, 3, 3, 20, 10}:");
		addStudentBeforeAnother(students, "name3", "name4", new int[] { 2, 3, 3, 20, 10 });

		System.out.println("Remove student with name3");
		deleteStudent(students, "name3");

	}
}
