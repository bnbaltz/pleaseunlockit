QUESTION:

"java" Write a program that reads the ages of three persons from the user, and decides who is the oldest, and who the youngest person is.

ANSWER:

#include<iostream>

using namespace std;

int main()

{

int age[3];

int num=1;

cout<<"How many persons are there in list <max - 10> ? ";

cin>>num;

for(int x=0; x <num; x++)

{

cout<<"Enter person"<<x+1<<" age: ";

cin>>age[x];

}

int eldest = age[0];

int young = age[0];

for(int y=1; y < num; y++)

{

if(age[y] > eldest)

{

eldest = age[y];

}

if(age[y] < young)

{

young = age[y];

}

}

cout<<"Age of eldest person is "<<eldest<<endl;

cout<<"Age of youngest person is "<<young<<endl;

return 0;

}
