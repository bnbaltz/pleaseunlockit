import java.util.Random;

public class BrandName {
	
	public static void main(String[] args) {
		
		
		System.out.println("Brand name generater - Your Name");
		for(int p=0 ;p <10; p++) {
			System.out.println(p+1+ ") "+ BrandName.generateName(7));
		}
		
	}
	
	public static String generateName(int size) {
		if(size < 6 || size > 12) {
			System.out.println("Enter size in range 6 to 12");
			return null;
		}
		/* initialising empty string to store brand name*/
		String name = "";
		
		/* array of vowel characters*/
		Character vowel[] = {'a','e','i','o','u'};
		
		/* array of consonant characters*/
		Character consonant[] = {'b', 'c', 'd', 'f', 'g', 'h', 'j', 'k', 'l', 'm', 'n', 'p', 'q', 'r', 's', 't', 'v', 'w', 'y', 'z'};
		
		// Creating random class object to generate random number in the range 
		Random rand = new Random(); 
		
		// loop to assign characters to name of given length/size
		for(int i=0; i<size; i++) {
			
			if( i ==0 ) {
				name = name + Character.toUpperCase(consonant[rand.nextInt(consonant.length)]);
			}
			if( i%2 ==0 ) {
				name = name + vowel[rand.nextInt(vowel.length)];
			}
			else {
				name = name + consonant[rand.nextInt(consonant.length)];
			}
		}
		
		return name;
	}

}
