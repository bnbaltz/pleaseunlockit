import operator
def CountCategories(movies):
    l1=[]

    #remove duplicate
    for i in movies:
        if i not in l1:
            l1.append(i)
            
    #counting number of movie
    l1.sort()
    l2=[]
    for i in l1:
        c=movies.count(i)
        l2.append(c)
    t=zip(l1,l2)
    import operator
    d=sorted(t,key=operator.itemgetter(1),reverse=True)
    #return dictionary
    return dict(d)

#driver code
movies=["Dark Knight","Harry Potter","The wall","Star war","Fast",
        "Dark Knight","Dark Knight","Star war","Star war","Harry Potter"]

result=CountCategories(movies)
dict(sorted(result.items(), key=lambda item: item[1]))
print(result)
