#include<stdio.h>
#include<math.h>

int main(){
    int initial_enrollment = 29933;
    int target;
    int valid_return;
    printf("Enter your enrollment target for year 2030: ");
    //Scanf will return positive integer value only when input is an integer
    valid_return = scanf("%d", &target);
    while (1){
        //Check if input was valid
        if (valid_return == 1 && target>0)
            break;
        //If input was not valid, give error message, and ask for input again
        else{
            printf("Invalid input. Try again.\n");
            printf("Enter your enrollment target for year 2030: ");
            fflush(stdin);
            valid_return = scanf("%d", &target);            
        }
    }

    //Calculate growth rate using given formula
    double growth_rate = pow(target*1.0/initial_enrollment, 1 / 11.0) - 1;
    growth_rate *= 100;        //To convert to percentage terms
    printf("Growht rate: %6.2lf\n", growth_rate);

    //Print evaluation message as per given condition
    if (growth_rate < 0)
        printf("Negative growth rate\n");
    else if (growth_rate >= 0 && growth_rate < 1)
        printf("Reasonable growth rate\n");
    else if (growth_rate <= 1 && growth_rate < 2)
        printf("Ambitious growth rate\n");
    else if (growth_rate <= 2 && growth_rate < 4)
        printf("Very ambitious growth rate\n");
    else if (growth_rate >= 4)
        printf("Unreasonable growth rate\n");

    //Base year
    int year = 2019;
    //Print header
    printf("%6s %10s\n", "Year", "Enrollment");
    //Calculate for each year and print
    while (year <= 2069){
        target = initial_enrollment * pow(1 + growth_rate/100, year - 2019);
        printf("%6d %10d\n", year, target);
        year++;
    }
    getch();
    return 0;
}
