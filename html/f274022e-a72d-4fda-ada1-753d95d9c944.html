import numpy as np
import matplotlib.pyplot as plt
#Read in the data to an array, called elevations:
elevations = np.loadtxt('elevationsNYC.txt')

mapShape = elevations.shape + (3,)

#Create a blank image that's all zeros:
floodMap = np.zeros(mapShape)

for row in range(mapShape[0]):
for col in range(mapShape[1]):
if elevations[row,col]>5 and elevations[row,col]<=20:#greater than5, less than 20
floodMap[row,col,:]=[.5,.5,.5]#set whole row together at once
elif elevations[row,col] <= 0:
#Below sea level
floodMap[row,col,2] = 1.0   
elif elevations[row,col] <= 6:
floodMap[row,col,0] = 1.0   
else:
floodMap[row,col,1] = 1.0   


#plt.imshow(floodMap)


#plt.show()

#Save the image:
plt.imsave('floodMap.png', floodMap)
